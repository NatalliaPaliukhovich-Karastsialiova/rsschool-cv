function toCamelCase(str){

    let res = ""

    for(let i = 0; i < str.length; i++){
        if(str[i] === "-" || str[i] === "_"){
        res += str[i + 1].toUpperCase()
        i++;
        } else {
        res += str[i];
        }
    }

    return res;
}
